35           } else {

37   
38           }

43           List<Veiculo> veiculoes = veiculoRepository.findAll();

45       }

47       public ResponseEntity<?> findAtivo( ) {
48           List<Veiculo> veiculos = veiculoRepository.findAll();
49           List<Veiculo> veiculosAtivos = new ArrayList<>();
50           veiculos.forEach(veiculo -> {if(veiculo.isAtivo()){
51               veiculosAtivos.add(veiculo);
52           }
53           });
54           return  ResponseEntity.ok().body(veiculosAtivos);
55       }
56   
57       public ResponseEntity<?> update(Long id, VeiculoDTOS veiculoDTOS) {
58           Optional<Veiculo> optionalVeiculo = veiculoRepository.findById(id);
59           if (optionalVeiculo.isEmpty()) {
60               return ResponseEntity.status(HttpStatus.NOT_FOUND).body("Veiculo not found with ID: " + id);
61           } else {
62               Veiculo veiculo = optionalVeiculo.get();
63               BeanUtils.copyProperties(veiculoDTOS, veiculo);
64               veiculo.setAtualizacao(LocalDateTime.now());
65               veiculoRepository.save(veiculo);
66               return ResponseEntity.ok().body("Veiculo atualizado com sucesso");
67           }
68       }
69   
70       @Transactional
71       public ResponseEntity<?> create(Veiculo veiculo) {
72   
73           if ( veiculoRepository.findByPlate(veiculo.getPlate()).isEmpty() ) {
74   
75   
76               try {
77                   veiculo.setAtivo(true);
78                   veiculoRepository.save(veiculo);
79                   TransactionAspectSupport.currentTransactionStatus().flush();
80                   return ResponseEntity.status(HttpStatus.CREATED).body(veiculo);
81               } catch (Exception e) {
82                   TransactionAspectSupport.currentTransactionStatus().setRollbackOnly();
83                   return ResponseEntity.badRequest().body(e.toString());
84               }
85           } else {
86               return ResponseEntity.status(HttpStatus.CONFLICT).body("placa ja cadastrada");
87           }
88       }
89   
90       public ResponseEntity<?> delete (Long id) {
91   
92           Optional<Veiculo> optionalVeiculo = this.veiculoRepository.findById(id);
93   
94           if(optionalVeiculo.isPresent()) {
95               Veiculo veiculo = veiculoRepository.getById(optionalVeiculo.get().getId());
96               if (movimentacaoRepository.findByVeiculoId(id).isEmpty()){
97   
98                   veiculoRepository.delete(veiculo);
99                   return ResponseEntity.ok().build();
100              } else {
101  
102                  veiculo.setAtivo(false);
103                  this.veiculoRepository.save(veiculo);
104                  return ResponseEntity.ok().build();
105              }
106          } else {
107              return ResponseEntity.notFound().build();
108          }
109      }
